Join data

We have

  • subscriptions available in a SAS temporary data set (imported from a CSV file)
  • reviews available in a JSON library (from a JSON file)
  • technical support evals available in an existing SAS data set
  • main customer churn data available in a Parquet library (pointing to Google Cloud Storage)
  • customer information available in a Snowflake data lakehouse
proc sql nowarn ;
   create table churn1                                                                       /* Create a table in WORK */
      (drop=custId customerSubscrCode reviewId ordinal_root ordinal_reviews) as           /* Drop unnecessary columns */           
      select *
      from gcs.customer_churn_data as churn                                                  /* GCS Parquet file */
         left join cust.customers as cust on churn.custId=cust.custId                        /* Snowflake table */
         left join subscriptions as subs on cust.customerSubscrCode=subs.customerSubscrCode  /* Import from CSV file */
         left join tcs.techSupportEvals as evals on churn.ID=evals.ID                        /* SAS data set */
         left join rev.reviews as rev on churn.reviewId=rev.reviewId                         /* Import from JSON file */
    ;
quit ;
